\section*{An Efficient DL-based PRF}

\textbf{Do not be intimidated by the long problem. It's broken down into four small parts and each part is quite doable. At the end of the four parts, you get a pretty cool theorem.}


The goal of the problem below is to show the following. Let $\G$ be a group of prime order $q$ generated by an element $g$. Let $k$ be some constant. Let $\alpha_0, \dots, \alpha_\ell$ be elements of $\G$. For an input $x$ that consists of $n$ bits $b_1, \dots, b_\ell$, let $F((\alpha_0, \dots, \alpha_\ell), x)$ be computed as follows:
\begin{enumerate}
\item Compute \[\alpha_x=\alpha_0 \cdot \left(\prod_{i \mbox{ such that }b_i = 1} \alpha_i \right)= \alpha_0 \cdot\left( \prod_{i =1}^\ell \alpha_i^{b_i}\right)\]
\item Output $g^\alpha_x$
\end{enumerate}

You will prove that $F((\alpha_0, \dots, \alpha_\ell), x)$ is a PRF, where $(\alpha_0, \dots, \alpha_\ell)$ is the key and $x$ is the input.

The big idea of the proof is the same as the Goldreich-Goldwasser-Micali binary tree construction of Section 4.6: this PRF can be viewed as a binary tree of depth $\ell$, and each level is a PRG. Here is how. Let $h=g^\alpha_0$.

\begin{itemize}
\item The root level contains one node $h$. 
\item The next level contains two nodes $h$ and $h^{\alpha_1}$. 
\item The next level contains four nodes $h, h^{\alpha_2}, h^{\alpha_1}, h^{\alpha_1\alpha_2}$
\item The next level contains eight nodes $h, h^{\alpha_3}, h^{\alpha_2}, h^{\alpha_2\alpha_3}, h^{ \alpha_1},h^{\alpha_1\alpha_3}, h^{\alpha_1\alpha_2},h^{\alpha_1\alpha_2\alpha_3}$.
\item \dots
\item The level at depth $d$ contains $g^{\alpha_x}$ for every binary string $x$ where the first $d$ bits are arbitrary and the last $\ell-d$ bits are 0.
\item\dots
\item The last level contains $g^{\alpha_x}$ for every for every binary string of length $\ell$
\end{itemize}

All we need to prove is that each level $d$ is a secure PRG when the corresponding $\alpha_d$ is kept secret. Thus, let $\prg:\G^k\times\Z_q\rightarrow \G^{2k}$ be defined as follows: on input $(X_1, \dots, X_k, \alpha)$, compute $Y_i=X_i^\alpha$ for $i=1\dots k$. Output $(X_1, Y_1, X_2, Y_2, \dots, X_k, Y_k)$.

\begin{theorem} Under the DDH assumption (Definition 10.8 in Section 10.5) on $\G$, $\prg$ is a secure pseudorandom generator.
\end{theorem}

You will now prove this theorem by proving the following parts.

\begin{problem}  (60 points at 15 each)

\begin{ppart}
Consider the following assumption we call \DDHprime. It is exactly the same as DDH described in Attack Game 10.6 and Definition 10.8, except in Experiment 1, $\gamma$ is chosen at random from $\Z_q-\{\alpha \beta\}$ (call this modified experiment $1'$ and the corresponding advantage $\DDHprimeadv$). Show that assuming DDH holds for $\G$, then so does \DDHprime. (\emph{Hint:} apply the difference lemma to prove that experiments 1 and 1' are indistinguishable.)
\end{ppart}

\begin{solution}
Let $\A$ be some adversary. 
Let $W_0$, $W_1$, and $W_{1'}$ be the events that $\A$ outputs $1$ in Experiment 0, Experiment 1, and Experiment $1'$ of the respective games.
Then, by definition:
\[
\DDHadv[\A] = \left|\Pr[W_0] - \Pr[W_1]\right| \quad \text{and} \quad \DDHprimeadv[\A] = \left|\Pr[W_0] - \Pr[W_{1'}]\right|.
\]

\noindent Now, we can couple Exp 1 and Exp $1'$ on the same underlying probability space.
In both, the challenger will sample $\alpha, \beta \rfrom \Z_q$, and then compute $u = g^\alpha, v = g^\beta$.
In Exp 1 specifically, $\gamma \rfrom \Z_q$. In Experiment $1'$, $\gamma \rfrom \Z_q \setminus \{\alpha\beta\}$.
Now, let's say that the ``bad event'', $Z$, is when $\gamma = \alpha\beta$ in Exp 1. Note that $\gamma$ is uniform in $\Z_q$ and is independent to $\alpha,\beta$, $\Pr[Z] = 1/q$.
Conditioned on $\neg Z$, the distributions in both experiments are identical. Thusly, by the difference lemma:
\[
\left|\Pr[W_1] - \Pr[W_{1'}]\right| \leq \Pr[Z] = \frac{1}{q}.
\]

\noindent And by the triangle inequality:
\[
\DDHprimeadv[\A] = \left|\Pr[W_0] - \Pr[W_{1'}]\right| \leq \left|\Pr[W_0] - \Pr[W_1]\right| + \left|\Pr[W_1] - \Pr[W_{1'}]\right| \leq \DDHadv[\A] + \frac{1}{q}.
\]

\noindent Assuming DDH holds for $\G$, then $\DDHadv[\A]$ is negligible. 
Since $q$ is exponential in the security parameter, $1/q$ is then also negligible. 
Thus, $\DDHprimeadv[\A]$ is negligible, proving $\text{DDH}'$ holds for $\G$.
\end{solution}

\begin{ppart}
Prove the following.
\begin{lemma}
Fix any $\alpha$, $\beta$, and $\gamma \not\equiv_q \alpha\beta$ in $\Z_q$. Define the map $M: \Z_q\times \Z_q \to \Z_q\times \Z_q$ as follows. Given inputs $d, e$:
\begin{align*}
x&=(d+\beta e)\bmod q\,;\\
y&=(\alpha d+\gamma e)\bmod q\,;\\
&\mbox{Output } (x, y)\,. 
\end{align*}
This map is a bijection.
\end{lemma}

\emph{Hint:} Find the inverse of $M$ by solving a system of two linear equations. Linear algebra extends to modular arithmetic modulo a prime $q$. Note that $\alpha, \beta$, and $\gamma$ are not inputs to the map; rather, they are used to specify a particular map.

\end{ppart}

\begin{solution}
Note the linear system mod, $q$, that's given:
\[
\begin{cases}
x \equiv d + \beta e \pmod q, \\
y \equiv \alpha d + \gamma e \pmod q.
\end{cases}
\]

\noindent Now, look at the first congruence. Solving for $d$ gets us:
\[
d \equiv x - \beta e \pmod q.
\]

\noindent And by subbing this into the 2nd congruence:
\[
y \equiv \alpha(x - \beta e) + \gamma e \equiv \alpha x + (\gamma - \alpha\beta)e \pmod q.
\]

\noindent Rearrange the terms:
\[
(\gamma - \alpha\beta)e \equiv y - \alpha x \pmod q.
\]

\noindent And since $\gamma \not\equiv_q \alpha\beta$ and $q$ is prime, $\gamma - \alpha\beta \not\equiv 0 \pmod q$.
In the field $\mathbb{Z}_q$, every non-0 element has a multiplicative inverse, so it is possible to solve uniquely for $e$:
\[
e \equiv (\gamma - \alpha\beta)^{-1}(y - \alpha x) \pmod q.
\]

\noindent And by subbing this expression for $e$ back into $d \equiv x - \beta e$, this gives us a unique value for $d$ mod $q$.
Thus, for every pair $(x,y) \in \mathbb{Z}_q \times \mathbb{Z}_q$, there exists exactly one pair $(d,e)$ that maps to it under $M$.
This makes an explicit inverse map $M^{-1}$, proving that $M$ is invertible. 
And since $M$ is a map on a finite set and is invertible, it is a bijection.
\end{solution}

\begin{ppart} 
Prove the following. (\emph{Hint:} work in the exponent space---i.e., consider logarithms of $X$ and $Y$ base $g$ --- and use the previous part. The variable naming here deliberately works well with the previous part.)
\begin{lemma}
For any $u=g^\alpha$, $v=g^\beta$, and $w_1 \ne g^{\alpha\beta}$, if $d$ and $e$ are a uniformly random pair of elements from $\Z_q$, then $X=g^d v^e$ and $Y=u^d w_1^e$ is a uniformly random pair of elements from $\G$ (i.e., every pair $X, Y$ occurs with probability $1/q^2$).
\end{lemma}
\end{ppart}

\begin{solution}
Let $\gamma = \log_g(w_1)$. 
Now, since $w_1 \neq g^{\alpha\beta}$, then we have $\gamma \not\equiv_q \alpha\beta$.
This is because $g$ generates $\G$ of prime order $q$, which is the discrete log map $z \mapsto \log_g(z)$, which is a bijection from $\G$ to $\Z_q$.\\

\noindent Now let $x = \log_g(X)$, and $y = \log_g(Y)$.
By subbing in $u=g^\alpha$, $v=g^\beta$, and $w_1=g^\gamma$ into the definitions of $X$ and $Y$, we get the following system mod $q$:
\[
\begin{cases}
x \equiv d + \beta e \pmod q, \\
y \equiv \alpha d + \gamma e \pmod q.
\end{cases}
\]

\noindent This is exactly the map $M: \Z_q \times \Z_q \to \Z_q \times \Z_q$ from Lemma 1 (1b).
And since $\gamma \not\equiv_q \alpha\beta$, (1b) guarantees that $M$ is a bijection.\\

\noindent Because $(d,e)$ is chosen uniformly at random from $\Z_q \times \Z_q$ and $M$ is a bijection, the image $(x,y) = M(d,e)$ must then also be uniformly distributed over $\Z_q \times \Z_q$.
By applying the bijection $z \mapsto g^z$ component-wise, it is then the case that $(X,Y) = (g^x, g^y)$ is uniformly distributed over $\G \times \G$.
Therefore, every pair $(X,Y) \in \G^2$ occurs with probability exactly $1/q^2$.
\end{solution}
\begin{ppart} 
Use the previous part to build a reduction that shows \DDHprime{} implies the security of $\prg$. (Combined with part (a), this proves Theorem 1.)
In particular, show that for every efficient adversary $\Adv$ breaking $\prg$ with advantage $\epsilon$, there is an efficient adversary $\Adv'$ breaking \DDHprime{} with advantage $\epsilon$ as well.
(\emph{Hint:} This is a single reduction; no hybrid arguments are needed. The previous part tells you exactly how your reduction should transform $u, v$, and $w$ that it receives from the \DDHprime{} challenger. It should that transformation $k$ times.)
\end{ppart}

\begin{solution}

Let $\Adv$ be the adv. that breaks the PRG $\prg$ with advantage $\epsilon$. 
Then, we can make another adv. $\Adv'$ that breaks the $\text{DDH}'$ assumption with the same advantage.\\

\noindent $\Adv'$ will get a challenge triple $(u, v, w)$ from the $\text{DDH}'$ challenger. $\Adv'$ will then do the following:
\begin{enumerate}
    \item $\Adv'$ takes $2k$ independent values $d_1, e_1, \dots, d_k, e_k \rfrom \Z_q$.
    \item For each $i \in \{1, \dots, k\}$, $\Adv'$ will calculate:
    \[
    X_i = g^{d_i} v^{e_i} \quad \text{and} \quad Y_i = u^{d_i} w^{e_i}.
    \]
    \item $\Adv'$ then sends the sequence $(X_1, Y_1, \dots, X_k, Y_k)$ to $\Adv$.
    \item $\Adv'$ will output whatever $\Adv$ outputs.
\end{enumerate}

\noindent Now, by analyzing the distribution of the sequence passed to $\Adv$ in the two Exp's of the $\text{DDH}'$ game:
\begin{itemize}
    \item \textbf{Exp 0 of $\text{DDH}'$:} Here, $u = g^\alpha$, $v = g^\beta$, and $w = g^{\alpha\beta}$ for some rnd $\alpha, \beta \rfrom \Z_q$.\\
    For each $i$, we have:
    \[
    Y_i = (g^\alpha)^{d_i} (g^{\alpha\beta})^{e_i} = g^{\alpha(d_i + \beta e_i)} = (g^{d_i + \beta e_i})^\alpha = X_i^\alpha.
    \]
    Now, since $d_i, e_i$ are rnd and independent, $X_i = g^{d_i} (g^\beta)^{e_i}$ is randomly distributed in $\G$. Thus, the sequence $(X_1, X_1^\alpha, \dots, X_k, X_k^\alpha)$ is distributed exactly as the output of $\prg$ on a uniformly random seed $(X_1, \dots, X_k, \alpha)$. This is in line with \textbf{Exp 0} of the PRG security game.
    
    \item \textbf{Exp $1'$ of $\text{DDH}'$:} Here, $u = g^\alpha$, $v = g^\beta$, and $w = g^\gamma$ where $\gamma \not\equiv_q \alpha\beta$.
    By Lemma 2 (1c), for fixed $u, v, w$ satisfying this condition, the pair $(X_i, Y_i) = (g^{d_i} v^{e_i}, u^{d_i} w^{e_i})$ is uniformly distributed over $\G \times \G$ when $d_i, e_i$ are rnd. Since the pairs are rnd, the entire sequence $(X_1, Y_1, \dots, X_k, Y_k)$ is uniformly distributed over $\G^{2k}$. This is in line with \textbf{Exp 1} of the PRG security game.
\end{itemize}

\noindent Now since $\Adv'$ perfectly simulates the PRG game for $\Adv$, then the advantage of $\Adv'$ is exactly equal to the advantage of $\Adv$:
\[
\DDHprimeadv[\Adv'] = \PRGadv[\Adv, \prg] = \epsilon.
\]
Thus, if $\text{DDH}'$ holds, $\prg$ is a secure PRG. And by (1a), this proves that $\prg$ is secure under the standard DDH assumption.

\end{solution}


\end{problem}



\section*{Backdoored PRGs}



\begin{problem} (40 points at 20 points each)

    In 2006, the National Institute for Standards and Technology (NIST) unveiled a new standard called \href{https://en.wikipedia.org/wiki/Dual_EC_DRBG}{Dual\_EC\_DRBG}, a PRG construction designed by the NSA. The construction was peculiar. It was clunkier than other existing PRGs, and up to 1000 times slower. Then, in 2007, two Microsoft cryptographers discovered something very unusual: the scheme had the possibility to contain a backdoor, which would allow an adversary to predict all future randomness given some output of the PRG. The existence of such a backdoor was confirmed by the \href{https://en.wikipedia.org/wiki/2010s_global_surveillance_disclosures}{Snowden disclosures} in 2013: the NSA had indeed inserted a backdoor into this PRG.

Dual\_EC\_DRBG is designed as a stateful PRG. At its core is a PRG $G$ which is repeatedly composed with itself sequentially, as in section 3.4.2 of the textbook. We know that this construction is secure as long as the underlying $G$ is a secure PRG.

The following is a simplified version of the Dual\_EC\_DRBG PRG\footnote{The real scheme uses an elliptic curve group instead of $QR_p$, and a different bijection is used. It also removes some bits of the PRG output. What this problem calls $g$ and $h$ are referred to as $Q$ and $P$.}. Let $p$ be a Sophie Germain prime, and $q=(p-1)/2$. Let $g$ and $h$ be generators of $QR_p$, sampled independently and uniformly. Let $\mathrm{abs}: QR_p \to \mathbb{Z}_q$ be the bijection defined in 8.5.1, such that $\mathrm{abs}(z)$ returns $z$ if $z<p/2$ and $-z$ otherwise.
The underlying PRG is as follows:

\begin{figure}[h!]
    
\centering 
\fcodebox{$G(s_0):$\\
\> $r \gets h^{s_0}$\\
\> $s_1 \gets \mathrm{abs}\left( g^{s_0} \right) $ \\
\> return $(r,s_{1})$
}
\end{figure}

\begin{ppart}
    Prove that when $g$ and $h$ are sampled uniformly at random, $G$ is a secure PRG, assuming DDH is hard in $QR_p$. To capture the notion of $g$ and $h$ being generated uniformly at random, we need to add a step to the PRG security game (Attack Game 3.1 in the textbook): we will have the challenger sample $g$ and $h$ and send them to the adversary as the first step, and then use those values when computing $G$ in Experiment 0.

\end{ppart}

\begin{solution}
Assume, for the sake of contradiction, $\A$ is an eff. adv. that can distinguish $G$ with non-negl. advantage $\epsilon$.
Then, it is possible to make another eff. adv. $\B$ that breaks $QR_p$ with the same advantage.\\

\noindent $\B$ will get $(u, v, w)$ from the DDH challenger, where $u=g^\alpha$, $v=g^\beta$, and $w$ is either $g^{\alpha\beta}$ (Exp 0) or $g^\gamma$ for rnd $\gamma$ (Exp 1).
$\B$ will then do the following:
\begin{enumerate}
    \item $\B$ sends $(u, v)$ to $\A$ as the public parameters $(g, h)$.
    \item $\B$ will calculate $r \gets w$ and $s_1 \gets \mathrm{abs}(u)$.
    \item $\B$ will then send the pair $(r, s_1)$ to $\A$.
    \item $\B$ outputs whatever $\A$ outputs.
\end{enumerate}

Now look at the distribution of $(r, s_1)$ passed to $\A$ in the two DDH experiments:
\begin{itemize}
    \item \textbf{Exp 0 (Real):} Here, $w = g^{\alpha\beta}$. Let the implicit PRG seed be $s_0 = \alpha$. Then:
    \[
    r = w = g^{\alpha\beta} = (g^\beta)^\alpha = v^\alpha = h^{s_0},
    \]
    and
    \[
    s_1 = \mathrm{abs}(u) = \mathrm{abs}(g^\alpha) = \mathrm{abs}(g^{s_0}).
    \]
    This is just the same output distribution as $G(s_0)$ in \textbf{Exp 0} of the PRG game.
    
    \item \textbf{Exp 1 (Rnd):} Here, $w = g^\gamma$ where $\gamma$ is chosen randly and independently of $\alpha, \beta$. 
    Thus, $r = w$ is uniformly distributed in $QR_p$. 
    Since $\alpha$ is uniform, $u = g^\alpha$ is also uniformly distributed in $QR_p$. \\\\
    Because $abs: QR_p \to Z_q$ is a bijection (from question, 8.5.1), $s_1 = abs(u)$ is uniformly distributed in $Z_q$. 
    Moreover, $r$ and $s_1$ are independent because $\gamma$ and $\alpha$ are independent. 
    Then, this just becomes the same uniform distribution as in \textbf{Exp 1} of the PRG game.
\end{itemize}

\noindent Now since $\B$ perfectly simulates the modified PRG security game for $\A$, the advantage of $\B$ just becomes that of $\A$:
\[
\DDHadv[\B] = \PRGadv[\A] = \epsilon.
\]
\noindent By our initial assumption, DDH is hard in $QR_p$, so $\DDHadv[\B]$ must be negligible.
Therefore, $\PRGadv[\A]$ is negligible, proving that $G$ is a secure PRG.
\end{solution}

\begin{ppart}
    Now show that if $g$ and $h$ are generated maliciously, the scheme has a backdoor. In particular, describe how to generate $g$ and $h$ in a way that produces a trapdoor key $sk$ and an efficient function $T$ that takes in $sk$ and $r_i$ (the $i$-th output of $G$ using $g$ and $h$, using an unknown random seed), and computes the next state $s_{i+1}$. (Note that once you know the state, you can compute all future outputs of the PRG.)

    Your maliciously generated $g$ and $h$ should be indistinguishable from an honestly generated $g$ and $h$. This ensures that the existence of an intentional backdoor is undetectable.

\end{ppart}

\begin{solution}
The NSA generator will do the following:
\begin{enumerate}
    \item Sample $g \rfrom QR_p$ uniformly at rnd.
    \item Sample $c \rfrom \Z_q$ uniformly at rnd.
    \item Set $h = g^c$.
    \item Output the public parameters $(g, h)$, while keeping the trapdoor key $sk = c$ secret.
\end{enumerate}

\noindent Now, given the trapdoor $sk = c$ and the $i$-th output $r_i = h^{s_i}$, the backdoor function $T$ should recover the next state $s_{i+1}$ as follows:
\[
T(sk, r_i) = \mathrm{abs}\left( r_i^{c^{-1} \bmod q} \right).
\]

\noindent \textbf{Correctness:} Since $r_i = h^{s_i} = (g^c)^{s_i} = g^{c s_i}$, then raising $r_i$ to the power of $c^{-1} \bmod q$ will result in:
\[
r_i^{c^{-1}} = \left(g^{c s_i}\right)^{c^{-1}} = g^{s_i}.
\]

\noindent By applying the bijection $\mathrm{abs}(\cdot)$ to $g^{s_i}$, one will recover $s_{i+1} = \mathrm{abs}(g^{s_i})$.
Thus, by knowing $sk$, one should be able to efficently predict all future states and outputs.\\

\noindent \textbf{Indistinguishability of Parameters.}
In the honest generation, $g$ and $h$ are sampled independently and uniformly from $QR_p$. In the NSA generation, $g$ is sampled rndly from $QR_p$, and then $h$ is computed as $g^c$, where $c$ is sampled uniformly from $\Z_q$.
Now because $QR_p$ is a cyclic group of prime order $q$, for any fixed $g \neq 1$, the map $c \mapsto g^c$ will be a bijection from $\Z_q$ to $QR_p$.
Therefore, when $c$ is uniform in $\Z_q$, $h = g^c$ will be uniformly distributed over $QR_p$, and is therefore statistically independent of $g$. \\

\noindent Consequently, the joint distribution of $(g, h)$ produced by the NSA generator is \emph{the same} to the joint distribution produced by the honest generator.
The statistical distance between the two distributions is just $0$, making them indistinguishable from one another.
An adv. will not be able to detect the presence of the backdoor without first even knowing that $sk$ exists.
\end{solution}


\end{problem}
